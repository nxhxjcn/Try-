local function FindFirstCustomer()
	local children = Workspace.Customers:GetChildren()
	for i=1,#children do
		local c = children[i]
		-- Use :FindFirstChild() for efficiency and safety
		if c:FindFirstChild("Head") and c:FindFirstChild("Humanoid") and c.Head.CFrame.Z<102 then
			local dialogInstance = c.Head:FindFirstChild("Dialog")
			if dialogInstance and dialogInstance:FindFirstChild("Correct") then
				-- Use math.abs() for efficient absolute value
				if (c.Humanoid.SeatPart and c.Humanoid.SeatPart.Anchored) or (c.Humanoid.SeatPart==nil and math.abs(c.Head.Velocity.Z) < 0.0001) then
					local dialog = dialogInstance.Correct.ResponseDialog or ''
					local order = "MountainDew"
					if dialog:sub(-8)=="instead." then
						dialog = dialog:sub(-30)
					end
					if dialog:find("pepperoni",1,true) then
						order = "PepperoniPizza"
					elseif dialog:find("sausage",1,true) then
						order = "SausagePizza"
					elseif dialog:find("cheese",1,true) then
						order = "CheesePizza"
					end
					return c,order
				end
			end
		end
	end
end

				d=f
			end
			if p and d then
				return p,d
			end
		end
	end
	return p,d
end
