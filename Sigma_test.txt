do
	local reg = (getreg or debug.getregistry)()
	for i = 1, #reg do
		local f = reg[i]
		if type(f) == "function" then
			for k, v in next, getupvalues(f) do
				if typeof(v) == "Instance" then
					if v.Name == "CashOut" then
						setupvalue(f, k, {MouseButton1Click = {wait = function() end, Wait = function() end}})
					elseif v.Name == "StickerName" then
						setupvalue(f, k, nil)
					end
				end
			end
			if tostring(getfenv(f)) == "Music" then
				local consts = getconstants(f)
				local loc = false
				for ci, c in next, consts do
					if c == "location changed" then
						loc = true
					elseif loc and c == "SendData" then
						setconstant(f, ci, "ExplodeString")
						break
					end
				end
			end
		elseif type(f) == "table" and rawget(f, "FireServer") and rawget(f, "BindEvents") then
			network = f
		end
	end
end

-- Helper functions (OPTIMIZED: Early exit patterns)
local function FindFirstCustomer()
	local children = customersFolder:GetChildren()
	for i = 1, #children do
		local c = children[i]
		local head = ffc(c, "Head")
		if not head then continue end
		
		local hum = ffc(c, "Humanoid")
		if not hum then continue end
		
		if head.CFrame.Z >= 102 then continue end
		
		local dialog = ffc(head, "Dialog")
		if not dialog then continue end
		
		local correct = ffc(dialog, "Correct")
		if not correct then continue end
		
		if not ((hum.SeatPart and hum.SeatPart.Anchored) or (hum.SeatPart == nil and (head.Velocity.Z ^ 2) ^ .5 < .0001)) then continue end
		
		local dialogText = correct.ResponseDialog or ''
		local order = "MountainDew"
		if dialogText:sub(-8) == "instead." then
			dialogText = dialogText:sub(-30)
		end
		if dialogText:find("pepperoni", 1, true) then

local function FindBoxes()
	local c, o, f
	local children = allBoxFolder:GetChildren()
	for i = 1, #children do
		local b = children[i]
		local hasPizza = ffc(b, "HasPizzaInside")
		local pizza = ffc(b, "Pizza")
		
		if not (hasPizza or pizza) then continue end
		
		if not c and b.Name == "BoxClosed" and not b.Anchored and not hasPizza.Value then
			c = b
		elseif not o and b.Name == "BoxOpen" and not b.Anchored and not pizza.Value then
			o = b
		elseif not f and ((b.Name == "BoxOpen" and pizza.Value) or (b.Name == "BoxClosed" and hasPizza.Value)) then
			f = b
		end
		
		if c and o and f then return c, o, f end
	end
	return c, o, f
end
